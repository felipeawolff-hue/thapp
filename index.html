SQL das tabelas

CREATE TABLE nomes_templo (
  id BIGSERIAL PRIMARY KEY,
  link TEXT NOT NULL,
  genero CHAR(1) NOT NULL,
  ordenanca TEXT NOT NULL,
  concluidas TEXT[] DEFAULT '{}',
  status TEXT DEFAULT 'disponivel',
  observacao TEXT,
  fs_id TEXT,
  usado_por_nome TEXT,
  usado_por_ala TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  usado_em TIMESTAMPTZ
);
CREATE TABLE usuarios_templo (
  id BIGSERIAL PRIMARY KEY,
  nome_completo TEXT NOT NULL,
  data_nascimento DATE,
  numero_ajuda TEXT,
  ala TEXT,
  telefone TEXT,
  tem_conta_sud BOOLEAN DEFAULT false,
  genero_escolhido CHAR(1),
  ordenanca_escolhida TEXT,
  nome_id BIGINT REFERENCES nomes_templo(id),
  created_at TIMESTAMPTZ DEFAULT NOW()
);
